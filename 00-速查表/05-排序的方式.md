
```java title=java fold
// Arrays
Arrays.sort(arr);  // 数组排序
// Collections
Collections.sort(list);  // 列表排序
list.sort(null);  // 自然排序
```

```go title=go fold
sort.Ints(arr)        // int数组
sort.Strings(arr)     // string数组
sort.Slice(arr, func(i, j int) bool {
    return arr[i] < arr[j]
})
```

```js title=javasript fold
arr.sort()              // 默认按字符串排序
arr.sort((a, b) => a - b)  // 数字升序
```

```php title=php fold
sort($arr);        // 升序
rsort($arr);       // 降序
usort($arr, fn($a, $b) => $a <=> $b);  // 自定义
```

```python title=python fold
arr.sort()              # 原地排序
sorted(arr)             # 返回新列表
sorted(arr, key=lambda x: x)  # 自定义
```

```groovy title=groovy fold
list.sort()             // 原地排序
list.sort { it }        // 自定义
```

```kotlin title=kotlin fold
arr.sort()              // 数组原地排序
list.sorted()           // 返回新列表
list.sortBy { it }      // 按字段排序
```

```c title=c fold
#include <stdlib.h>
qsort(arr, n, sizeof(int), compare);
// compare函数: int compare(const void *a, const void *b)
```

```cpp title=c++ fold
#include <algorithm>
std::sort(arr.begin(), arr.end());
std::sort(arr, arr + n);  // 数组
```

```dart title=dart fold
list.sort();                    // 原地排序
list.sort((a, b) => a.compareTo(b));
```

```scala title=scala fold
arr.sorted                      // 返回新数组
list.sortBy(x => x)            // 按字段排序
list.sortWith(_ < _)           // 自定义
```

```ruby title=ruby fold
arr.sort                        # 返回新数组
arr.sort!                       # 原地排序
arr.sort { |a, b| a <=> b }    # 自定义
```

```rust title=rust fold
vec.sort();                     // 原地排序
vec.sort_by(|a, b| a.cmp(b));  // 自定义
```
